```typescript
  // Metadata

  priority        Int      @default(0)
  tags            String[]
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  // Relations
  usageRecords    AgentAttachmentUsage[]

  @@map("agent_attachments")
}

  id             String   @id @default(uuid())

  attachmentId   String

  relevanceScore Float    @default(0.5)
  usedAt         DateTime @default(now())

  // Relations
  attachment     AgentAttachment @relation(fields: [attachmentId], references: [id], onDelete: Cascade)

  @@map("agent_attachment_usage")
```

```typescript
// Enhanced migration for 100 agents
export class HistoricalAgentsService {
  static async migrate100Agents(): Promise<{
    success: boolean
    migrated: number
    categories: {
      ancient: number
      medieval: number
      enlightenment: number
      modern: number
      monica: number
    }
    errors: string[]
  }> {
    const results = {
      success: true,
      migrated: 0,
      categories: { ancient: 0, medieval: 0, enlightenment: 0, modern: 0, monica: 0 },
      errors: [] as string[],
    }

    // Migrate Monica first
    try {
      await this.migrateAgent(MONICA_AS_CRAFTED_AGENT, 'monica')
      results.categories.monica = 1
      results.migrated++
    } catch (error) {
      results.errors.push(`Monica migration failed: ${error}`)
      results.success = false
    }

    // Migrate by category for better organization
    const categories = [
      { agents: ANCIENT_AGENTS, name: 'ancient' },
      { agents: MEDIEVAL_AGENTS, name: 'medieval' },
      { agents: ENLIGHTENMENT_AGENTS, name: 'enlightenment' },
      { agents: MODERN_AGENTS, name: 'modern' },
    ]

    for (const category of categories) {
      for (const agent of category.agents) {
        try {
          await this.migrateAgent(agent, category.name)
          results.categories[category.name as keyof typeof results.categories]++
          results.migrated++
        } catch (error) {
          results.errors.push(`${agent.name} migration failed: ${error}`)
          results.success = false
        }
      }
    }

    return results
  }

  private static async migrateAgent(agent: CraftedAgent, category: string): Promise<void> {
    // Check if already exists
    const existing = await prisma.historicalAgent.findUnique({
      where: { id: agent.id },
    })

    if (existing) {
      console.log(`Agent ${agent.name} already exists, skipping...`)
      return
    }

    // Create with category metadata
    await prisma.historicalAgent.create({
      data: {
        id: agent.id,
        name: agent.name,
        title: agent.title,
        era: agent.background.era,
        culture: agent.background.culture,
        birthDate: agent.birthData.date,
        birthLocation: agent.birthData.location,
        consciousness: agent.consciousness,
        personality: agent.personality,
        abilities: agent.abilities,
        background: agent.background,
        monicaConstant: agent.consciousness.monicaConstant,
        monicaCreationStory: agent.monicaCreationStory,
        category: category, // Add category for organization
        isActive: true,
        enhancementStatus: 'personality_complete',
      },
    })
  }
}
```

**If System Performance Degrades:**

1. **Implement Agent Pagination:** Load agents in batches of 25
2. **Reduce Initial Load:** Show only featured agents initially
3. **Optimize Database Queries:** Add indexes and query optimization
4. **Implement Lazy Loading:** Load agent details on demand
5. **Use CDN Caching:** Cache agent profiles for faster access

**If Memory Usage Exceeds Limits:**

1. **Split Agent Data:** Move detailed profiles to separate files
2. **Implement Virtual Lists:** Only render visible agents
3. **Use Compression:** Compress large consciousness profiles
4. **Optimize Images:** Reduce avatar and symbol file sizes
5. **Implement Memory Cleanup:** Clear unused agent data

**If Legal Issues Arise:**

1. **Replace Problematic Agents:** Substitute with confirmed public domain figures
2. **Add Legal Disclaimers:** Educational and historical use statements
3. **Implement Age Filters:** Remove any figures died after 1950
4. **Documentation Review:** Ensure all content is factual and respectful
5. **Legal Consultation:** Seek expert review if needed

This comprehensive implementation guide provides detailed specifications, examples, workflows, and contingency plans for successfully expanding the Gallery of Perpetuity to 100 historical agents while maintaining quality, performance, and legal compliance throughout the development process.
